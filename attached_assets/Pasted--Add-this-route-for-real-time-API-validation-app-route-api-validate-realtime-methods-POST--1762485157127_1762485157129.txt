@app.route('/api/validate-realtime', methods=['POST', 'OPTIONS'])
def validate_realtime():
    # Handle CORS for cross-origin requests
    if request.method == 'OPTIONS':
        response = jsonify({'status': 'ok'})
        response.headers.add('Access-Control-Allow-Origin', '*')
        response.headers.add('Access-Control-Allow-Headers', 'Content-Type')
        response.headers.add('Access-Control-Allow-Methods', 'POST')
        return response
    
    try:
        data = request.get_json()
        phone = data.get('phone')
        country = data.get('country', 'US')
        
        if not phone:
            return jsonify({'error': 'Phone number required'}), 400
        
        # Call Veriphone API
        response = requests.get(
            'https://api.veriphone.io/v2/verify',
            params={
                'phone': phone,
                'key': VERIPHONE_API_KEY,
                'default_country': country
            },
            timeout=10
        )
        
        data = response.json()
        
        result = {
            'valid': data.get('phone_valid', False),
            'phone_type': data.get('phone_type', 'unknown'),
            'can_receive_sms': data.get('phone_type') == 'mobile',
            'carrier': data.get('carrier', 'Unknown'),
            'formatted': data.get('international_number', phone),
            'local_format': data.get('local_format', ''),
            'country': data.get('country_code', country),
            'warnings': []
        }
        
        # Add warnings
        if not result['valid']:
            result['warnings'].append('Invalid phone number')
        elif result['phone_type'] == 'fixed_line':
            result['warnings'].append('Landline - cannot receive SMS')
        elif result['phone_type'] == 'voip':
            result['warnings'].append('VOIP number - SMS delivery may be unreliable')
        
        response = jsonify(result)
        response.headers.add('Access-Control-Allow-Origin', '*')
        return response
        
    except Exception as e:
        error_response = jsonify({'error': str(e), 'valid': False})
        error_response.headers.add('Access-Control-Allow-Origin', '*')
        return error_response, 500
